get_torvik_advstats <- function(season_year, cn = cn67) {
  url <- sprintf("https://www.barttorvik.com/getadvstats.php?year=%d", season_year)

  txt <- request(url) |>
    req_user_agent("cbb-analytics (personal use)") |>
    req_perform() |>
    resp_body_string()

  rows <- fromJSON(txt, simplifyVector = FALSE)

  df <- rbindlist(lapply(rows, as.list), fill = TRUE, use.names = FALSE) |>
    as.data.frame(stringsAsFactors = FALSE)

  stopifnot(ncol(df) == length(cn))
  names(df) <- cn

  df %>%
    select(-starts_with("DELETE"))
}

model_table <- model_table_START %>%
  mutate(
    year = as.integer(year),
    pid = as.character(pid),
    player = str_squish(player),
    team = str_squish(team),

    conf = str_squish(conf),
    conf = str_to_lower(conf),
    conf = if_else(conf == "ind", "independent", conf),

    major = case_when(
      conf %in% c("b12", "acc", "b10", "sec", "be", "p12") ~ "high",
      TRUE ~ "mid"
    ),

    conf = factor(conf),
    major = factor(major, levels = c("high", "mid")),

    class = str_squish(class),
    class = na_if(class, ""),
    class = na_if(class, "N/A"),
    class = na_if(class, "--"),
    class = case_when(
      class %in% c("Fr", "So", "Jr", "Sr") ~ class,
      TRUE ~ NA_character_
    ),

    Role = str_squish(Role),
    Role = na_if(Role, ""),

    Birthdate = as.character(Birthdate),

    underclass = case_when(
      class %in% c("Fr", "So") ~ "Underclassmen",
      class %in% c("Jr", "Sr") ~ "Upperclassmen",
      TRUE ~ NA_character_
    ),
    
    player_team_gap = BPM - barthag
  ) %>%
  group_by(team, year) %>%
  mutate(
    mp_rank = min_rank(desc(MP)),
    bpm_rank = min_rank(desc(BPM)),
    usage_rank = min_rank(desc(USG))
  ) %>%
  ungroup() %>%
  group_by(conf, year) %>%
  mutate(
    bpm_pct_conf = percent_rank(BPM),
    porpag_pct_conf = percent_rank(PORPAG)
  ) %>%
  ungroup() %>%
  group_by(major, year) %>%
  mutate(
    usg_rank_c1 = min_rank(desc(USG)),
    porpag_rank_c1 = min_rank(desc(PORPAG)),
    bpm_rank_c1 = min_rank(desc(BPM))
  ) %>%
  ungroup() %>%
  mutate(
    underused_star = if_else(BPM > 5 & mp_rank > 5, 1, 0)
  ) %>%
  group_by(team, year) %>%
  mutate(
    team_total_mp = sum(MP, na.rm = TRUE),
    minutes_share = MP / team_total_mp,
    mp_pct_team = percent_rank(MP),
    bpm_pct_team = percent_rank(BPM),
    usg_pct_team = percent_rank(USG)
  ) %>%
  ungroup() %>%
  mutate(
    bpm_minutes_gap = BPM - minutes_share,
    mid_major_star = if_else(major == "mid" & bpm_pct_conf > 0.9, 1, 0),
    low_role_high_output = if_else(porpag_pct_conf >= 0.75 & mp_pct_team <= 0.40, 1L, 0L),
    mid_major_star = if_else(major == "mid" & bpm_pct_conf >= 0.90, 1L, 0L),
    high_major_low_role = if_else(major == "high" & mp_pct_team <= 0.25, 1L, 0L),

    winning_team = if_else(!is.na(team_wins) & !is.na(team_losses) & team_wins > team_losses, 1L, 0L),
    conf_winning_team = if_else(
      !is.na(conf_wins) & !is.na(conf_losses) & conf_wins > conf_losses,
      1L, 0L
    )
  )

next_year_lookup <- model_table %>%
  select(
    pid, year, player, Birthdate,
    next_team = team,
    next_conf = conf,
    next_class = class,
    next_role = Role
  ) %>%
  distinct() %>%
  mutate(year = year - 1)

transfer_table <- model_table %>%
  left_join(
    next_year_lookup,
    by = c("pid", "year", "player", "Birthdate")
  ) %>%
  mutate(
    returned_next_year = if_else(!is.na(next_team), 1L, 0L),
    transfer_next_year = case_when(
      is.na(next_team) ~ NA_integer_,
      next_team != team ~ 1L,
      next_team == team ~ 0L
    ),
    transfer_status = case_when(
      is.na(next_team) ~ "no_next_year_row",
      next_team != team ~ "transferred",
      next_team == team ~ "stayed"
    )
  )

boot_pred <- function(data, indices) {
  d <- data[indices,]

  fit <- glm(
    transfer_next_year ~ GP + ORTG + USG + ORB. + TP. +
      major*PORPAG + adjoe*adjde + `Conf Win%` +
      Role + class + MP*BPM +
      major*BPM +
      player_team_gap*bpm_minutes_gap +
      mid_major_star*underused_star,
    data = d,
    family = binomial()
  )

  predict(fit, newdata = valid_model, type = "response")
}

boot_results <- boot(

top_k_accuracy <- function(probs, actual, k = 3) {
  probs <- as.data.frame(probs)

  topk <- apply(probs, 1, function(x) {
    names(sort(x, decreasing = TRUE))[1:k]
  })

  # transpose so each row = one observation
  topk <- t(topk)

  mean(sapply(1:nrow(topk), function(i) {
    actual[i] %in% topk[i, ]
  }))
}

log_loss <- function(actual, probs) {
  probs <- as.matrix(probs)
  actual <- as.factor(actual)

  # create one-hot encoding
  actual_mat <- model.matrix(~ actual - 1)

  # align column names
  colnames(actual_mat) <- levels(actual)

  # reorder probs to match
  probs <- probs[, colnames(actual_mat)]

  eps <- 1e-15
  probs <- pmax(pmin(probs, 1 - eps), eps)

  -mean(rowSums(actual_mat * log(probs)))
}

  filter(!is.na(conf_tier_next)) %>%
  filter(!is.na(BPM), !is.na(PORPAG), !is.na(MP), !is.na(USG),
         !is.na(adjoe), !is.na(adjde), !is.na(barthag)) %>%
  filter(class %in% c("Fr", "So", "Jr", "Sr"))

rmse_fun <- function(actual, pred) {
  sqrt(mean((actual - pred)^2, na.rm = TRUE))
}

mae_fun <- function(actual, pred) {
  mean(abs(actual - pred), na.rm = TRUE)
}

r2_fun <- function(actual, pred) {
  ss_res <- sum((actual - pred)^2, na.rm = TRUE)
  ss_tot <- sum((actual - mean(actual, na.rm = TRUE))^2, na.rm = TRUE)
  1 - ss_res / ss_tot
}

eval_metrics <- function(actual, pred) {
  tibble(
    RMSE = rmse_fun(actual, pred),
    MAE  = mae_fun(actual, pred),
    R2   = r2_fun(actual, pred)
  )
}
